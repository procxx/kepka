        AppKit
        CoreWLAN
        IOKit
    )
        set(FW_LIB FW_LIB-NOTFOUND)
        find_library(FW_LIB ${framework})
        message(STATUS "Found framework ${FW_LIB}")
        target_link_libraries(Kepka ${FW_LIB})
    endforeach(framework)
    set(FW_LIB FW_LIB-NOTFOUND)
endif()
